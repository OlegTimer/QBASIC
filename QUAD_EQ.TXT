CLS
PRINT "CALCULATE QUARDRATIC EQUATION IN REAL NUMS"
PRINT "A*X^2 + B*X + C = 0"
PRINT
PRINT "PLEASE, ENTER INTEGER A. NOT 0"
INPUT A
PRINT "PLEASE, ENTER INTEGER B"
INPUT B
PRINT "PLEASE, ENTER INTEGER C"
INPUT C
DIM D#, X1#, X2#
D# = B * B - 4 * A * C
IF D# < 0 THEN
PRINT "NO REAL ROOTS": END
ELSE
X1# = (-1 * B + D# ^ .5) / (2 * A)
X2# = (-1 * B - D# ^ .5) / (2 * A)
PRINT "ROOTS ARE:"
PRINT X1#
PRINT X2#
END IF
END
